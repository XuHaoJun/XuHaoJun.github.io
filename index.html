帕斯卡三角形 (Ruby) Sep 3, 2013

帕斯卡三角形

def pascal_triangle(n, i = 2, result = [[1], [1,1]])
  if n == 1
    [[1]]
  elsif n == 2
    [[1], [1,1]]
  elsif i == n
    result
  else
    current_row = [1]
    (i-1).times do |n|
      current_row << (result[i-1][n] + result[i-1][n+1])
    end
    current_row << 1

    pascal_triangle(n, i+1, result << current_row)
  end
end

def get_uniq_nums(size, range)
  ary = []
  begin
    r = rand(range)
    if ary.member?(r)
      next
    else
      ary << r
    end
  end while( ary.length < size )
  return ary
end
